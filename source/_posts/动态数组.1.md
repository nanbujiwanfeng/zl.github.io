#include <stdio.h>
#include <stdlib.h>
int main()
{
    int n = 5;
    //申请一块连续的内存，字节总数为：20个字节
    int* ptr = (int*)malloc(n * sizeof(int)); 
    if (ptr != NULL)
    {
        for (int i = 0; i < n; i++)
        {
            //通过int类型指针每次访问4个字节
            *(ptr + i) = i + 1;
        }
        for (int i = 0; i < n; i++)
        {
            printf("%d ", *(ptr+i));
        }
    }
    return 0;
}
